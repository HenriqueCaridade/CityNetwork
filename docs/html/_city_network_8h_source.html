93 // Haversine Formula
94 if (other.lat == INFINITY || other.lon == INFINITY) return INFINITY; // Invalid.

97 const double radLat2 = other.lat * M_PI / 180;
98 const double deltaLat = radLat2 - radLat1;
99 const double deltaLon = (other.lon - this->lon) * M_PI / 180;

102 double aux = sinLat * sinLat + cos(radLat1) * cos(radLat2) * sinLon * sinLon;
