109 constexpr double operator-(const Node& other) const {
110 // Haversine Formula
111 if (other.lat == INFINITY || other.lon == INFINITY) return INFINITY; // Invalid.
112 if (this->lat == INFINITY || this->lon == INFINITY) return INFINITY; // Invalid.
113 const double radLat1 = this->lat * M_PI / 180;
114 const double radLat2 = other.lat * M_PI / 180;
115 const double deltaLat = radLat2 - radLat1;
116 const double deltaLon = (other.lon - this->lon) * M_PI / 180;
117 const double sinLat = sin(deltaLat / 2);
118 const double sinLon = sin(deltaLon / 2);
119 double aux = sinLat * sinLat + cos(radLat1) * cos(radLat2) * sinLon * sinLon;
120 aux = 2.0 * atan2 (sqrt(aux), sqrt(1.0 - aux));
121 return 6371000 * aux;
122 }
